312void Scheduler(int reset_count){
313
314 // bounds checking -> error msgs
315 if(reset_count < 0 || reset_count > MAX_COUNT || planner.analog_t0 > reset_count){
316 if(DEBUG) Serial.println("[ERROR] over ran max reset count without reseting count, check if variable is being reset or if tasks are not completing");

318 reset_count = planner.analog_t0 = planner.ht_t0 = planner.ota_t0 = planner.tlm_t0 = 1;
319 gator_prefs.putInt("reset_count", reset_count);
320 gator_prefs.putInt("analog_t0", planner.analog_t0);
321 gator_prefs.putInt("ht_t0", planner.ht_t0);
322 gator_prefs.putInt("ota_t0", planner.ota_t0);

324 }
325
326 bool run_vwc = reset_count - planner.analog_t0 >= VWC_FREQ;
327 bool run_ht = reset_count - planner.ht_t0 >= HT_FREQ;
328 bool run_ota_update = reset_count - planner.ota_t0 >= OTA_FREQ;
329 bool run_tlm = reset_count - planner.tlm_t0 >= TLM_FREQ;
330
331 if(run_vwc){
332 ReadWired();
333 planner.analog_t0 = reset_count;
334 gator_prefs.putInt("analog_t0", planner.analog_t0);
335 mqtt_client.loop();
336 }
337
338 if(run_ht){
339 ReadHT();
340 planner.ht_t0 = reset_count;
341 gator_prefs.putInt("ht_t0", planner.ht_t0);
342 // Publish happens in callback function
343 mqtt_client.loop();
344 }
345
346 if(run_ota_update){
347 planner.ota_t0 = reset_count;
348 gator_prefs.putInt("ota_t0", planner.ota_t0);
349 OTAUpdate();
350 mqtt_client.loop();
351 }
352
353 if(run_tlm){
354 SendTLM();
355 planner.tlm_t0 = reset_count;
356 gator_prefs.putInt("tlm_t0", planner.tlm_t0);
357 mqtt_client.loop();
358 }
