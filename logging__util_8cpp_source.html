57bool init_data_logger(){
58 Serial.println("[DEBUG] initializing new SDLogger");
59 // initialize SDLogger
60 logger = new SDLogger();
61
62 // check if the uSD card is connected
63 if(!logger->initialize_sd_card()){
64 return false;
65 }
66
67 // retrieve latest timestamp
68 std::string ts = cache_retrieve_timestamp();
69 std::string fn = "";
70 Serial.printf("[DEBUG] retrieved cached timestamp -> \'%s\'\n", ts.c_str());
71 if(ts != "" && !absolute_timestamp_available){
72 // 1. there was a cached timestamp
73 // 2. no absolute timestamp from NTP available
74 // therefore,
75 // use cached timestamp
76 tsp = new TimeStampParser(ts);
77 // generate log filename
78 fn = get_log_filename(tsp);
79
80 }else if(absolute_timestamp_available){
81 // use absolute timestamp to initialize
82 tsp = new TimeStampParser(tsb->get_date_time());
83 Serial.printf("[DEBUG] using absolute timestamp -> \'%s\'\n", tsp->get_timestamp().c_str());
84 fn = get_log_filename(tsp);
85
86 }else{
87 // 1. no cached timestamp
88 // 2. no absolute timestamp
89 // therefore,
90 // set to default log file name
91 fn = "/log";
92 // set timestamp to default value
93 tsp = new TimeStampParser("00-00-00T00:00:00+0");
94 Serial.printf("[DEBUG] using default timestamp -> \'%s\'\n", tsp->get_timestamp().c_str());
95 }
96
97 // ready to rock!
98 logger->set_filename(fn + ".csv");
99
100 return true;
101}
